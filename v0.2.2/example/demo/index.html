struct CornerPlayer <: Player end
function Reversi.get_move(player::CornerPlayer, game::ReversiGame)
    moves = valid_moves(game)
    if isempty(moves)
        return nothing
    end
    corners = [Position(1, 1), Position(1, 8), Position(8, 1), Position(8, 8)]
    for corner in corners
        if corner in moves
            return corner
        end
    end
    return rand(moves)
end
